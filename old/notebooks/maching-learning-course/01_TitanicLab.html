def plot_survivors_per_class():
    classes = df['Pclass']
    survived = (df['Survived'] == 1)
    unique_classes = np.unique(classes)
    survived_per_class = [np.logical_and(classes == c, survived).sum() for c in unique_classes]
    plt.bar(unique_classes - .4, survived_per_class, label='survived', color='green', alpha=.4)
    not_survived_per_class = [np.logical_and(classes == c, ~survived).sum() for c in unique_classes]
    plt.bar(unique_classes - .4, not_survived_per_class, label='not survived', bottom=survived_per_class, color='red', alpha=0.4)
    plt.xticks(unique_classes, ['{} class'.format(c) for c in unique_classes])
    plt.ylabel('Passangers count')
    plt.legend(loc='upper left')
