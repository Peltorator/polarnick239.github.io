x_old = np.zeros(b.shape[0])
r_old = b - A.dot(x_old)
z_old = r_old

steps = A.shape[0]
draw_isolines = (A.shape[0] == 2)

if draw_isolines:
    count = 100
    xs, ys = np.mgrid[0:count, 0:count] / count
    if solution is not None:
        xs, ys = xs.ravel() * solution[0] * 2, ys.ravel() * solution[1] * 2
    else:
        xs, ys = xs.ravel() * solution[0] * 100 * 2 - 50, ys.ravel() * 100 * 2 - 50
    plt.contour(xs.reshape(count, count), ys.reshape(count, count),
                ((b - A.dot([xs, ys]).T) * (b - A.dot([xs, ys]).T)).sum(axis=1).reshape(count, count),
                levels=np.power(np.arange(0.1, 50.0, 0.4), 2.0));

for i in range(3 * steps):
    print('Step {}...'.format(i + 1))
    alpha = (r_old * r_old).sum() / (A.dot(z_old) * z_old).sum()
    x = x_old + alpha * z_old
    r = r_old - alpha * A.dot(z_old)
    if (np.sum(r * r) / np.sum(b * b) < 1e-32):
        break
    beta = (r * r).sum() / (r_old * r_old).sum()
    z = r + beta * z_old
    if draw_isolines:
        plt.plot([x_old[0], x[0]], [x_old[1], x[1]], color='red', linestyle='-', linewidth=1)
    x_old, r_old, z_old = x, r, z

print('x0 = {:.6f}'.format(x[0]), end='')

    print(', x{} = {:.6f}'.format(i+1, x[i+1]), end='')
